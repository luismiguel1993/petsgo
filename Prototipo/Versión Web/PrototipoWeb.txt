            <div className="w-[410px] h-[860px] bg-[#0c0c0c] rounded-[80px] p-6 shadow-[0_80px_160px_-40px_rgba(0,168,232,0.4)] border-[14px] border-[#222] relative overflow-hidden">
              <div className="absolute top-0 left-1/2 -translate-x-1/2 w-36 h-9 bg-[#222] rounded-b-[30px] z-[1000]"></div>
              <MobileApp openBot={() => setIsBotOpen(true)} />
            </div>
            <div className="absolute -bottom-20 left-1/2 -translate-x-1/2 w-[150%] h-[25%] bg-cyan-500/15 blur-[150px] -z-10 rounded-full"></div>
          </div>
        )}
      </div>

      <BotChatOverlay isOpen={isBotOpen} onClose={() => setIsBotOpen(false)} />

      <button 
        onClick={() => setIsBotOpen(true)}
        style={{ backgroundColor: COLORS.secondary }}
        className={`fixed bottom-12 right-12 w-24 h-24 rounded-full shadow-[0_30px_60px_-10px_rgba(255,196,0,0.6)] flex items-center justify-center text-yellow-900 border-[8px] border-white z-[20000] active:scale-90 transition-all hover:rotate-12 group ${isImmersive && !isBotOpen ? 'scale-75 opacity-30 hover:opacity-100 hover:scale-100' : ''}`}
      >
        <div className="relative">
          <PawPrint size={48} className="group-hover:animate-bounce" />
          <div className="absolute -top-1 -right-1 w-6 h-6 bg-red-500 rounded-full border-4 border-white animate-pulse"></div>
        </div>
      </button>
    </div>
  );
};

export default App;
